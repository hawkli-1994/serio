serio

Modern, idiomatic Go serial port library

Effortless, context-aware, stream-friendly serial communication

Features

io.ReadWriteCloser interface

Use Go's standard IO tools (io.Copy, bufio, etc.)

Context support

Open, read, write operations support context for timeout/cancel.

Modern configuration

Easy, readable config with Config struct.

Cross-platform

Works on Linux, Windows, macOS, ARM.

Automatic resource management

Safe, defer-friendly resource handling.

Timeout control

Granular control over read/write timeouts with SetWriteTimeout and SetDeadline.

Quick Start

go get github.com/hawkli-1994/serio
package main

import (
    "context"
    "fmt"
    "log"
    "time"

    "github.com/hawkli-1994/serio"
)

func main() {
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()

    port, err := serio.Open(ctx, serio.Config{
        Name:     "/dev/ttyUSB0",
        Baud:     115200,
        DataBits: 8,
        StopBits: 1,
        Parity:   serio.None,
        Timeout:  time.Second * 2,
    })
    if err != nil {
        log.Fatal(err)
    }
    defer port.Close()

    _, err = port.Write([]byte("hello serial!"))
    if err != nil {
        log.Fatal(err)
    }

    buf := make([]byte, 128)
    n, err := port.Read(buf)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("Received: %s\n", buf[:n])
}
